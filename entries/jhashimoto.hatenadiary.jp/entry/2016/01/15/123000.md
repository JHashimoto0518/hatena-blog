String.Formatで "{" と "}" を出力するには、"{" または "}" を２つ続けて記述します。

====

>|cs|
using System;

namespace Sandbox {
    class Program {
        static void Main(string[] args) {
            string s = string.Format("{0}{{fuga}}", "hoge");
            Console.WriteLine(s);   // hoge{fuga} 
            Console.ReadKey();
        }
    }
}
