**ソースコード
>|cs|
using System;
using System.Linq;

namespace LinqSample {
    class Program {
        static void Main(string[] args) {
            int[] numbers = new int[] { 1, 2, 3, 4, 5 };
            Console.WriteLine(numbers.First(number => 3 < number));
            Console.ReadKey();
        }
    }
}
